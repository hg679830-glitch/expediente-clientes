Abre el Google Sheet donde se guardan los datos. * 2. Ve a Extensiones > Apps Script. * 3. Borra el contenido de Code.gs y pega TODO este archivo. * 4. Guarda el proyecto (icono de disquete). * 5. La primera vez que despliegues, Google puede volver a pedir autorizar * permisos (es tu propio script, es seguro aceptar). */ // Encabezados de la hoja, en el mismo orden que los campos del formulario. const HEADERS = [ "Fecha y hora de registro", "Asesor que captura", "Nivel de lealtad", "Fecha de captura", "Nombre completo", "Cómo le gusta que le llamen", "Fecha de nacimiento", "Estado civil", "Cónyuge o pareja", "Cumpleaños del cónyuge", "Aniversario de matrimonio", "Etapa educativa de los hijos", "Hijos: nombres y edades", "Escuela o universidad de los hijos", "Cliente de La Unión desde", "Aniversario de fundación de su empresa", "Giro o industria", "Número de empleados aprox.", "Facturación aproximada", "Canal preferido", "Horarios de contacto", "Frecuencia deseada", "Tono preferido", "Personas autorizadas", "Pasatiempos y aficiones", "Equipos o deportes favoritos", "Restaurantes y bebidas preferidas", "Destinos de viaje favoritos", "Preferencias de regalo", "Momentos de vida del negocio", "Cambios en la familia", "Referidos que ha dado" ]; // Mapea cada encabezado al "name" del campo correspondiente en el HTML. const FIELD_ORDER = [ null, // Fecha y hora de registro -> se llena automáticamente con new Date() "asesor", "nivel_lealtad", "fecha_captura", "nombre_completo", "como_le_gusta_llamar", "fecha_nacimiento", "estado_civil", "conyuge_nombre", "conyuge_cumpleanos", "aniversario_matrimonio", "hijos_etapa_educativa", "hijos_detalle", "hijos_escuela", "aniversario_cliente_union", "aniversario_fundacion_empresa", "giro_industria", "num_empleados", "facturacion_aproximada", "canal_preferido", "horarios_contacto", "frecuencia_deseada", "tono_preferido", "personas_autorizadas", "pasatiempos", "equipos_deportes_favoritos", "restaurantes_bebidas_preferidas", "destinos_viaje_favoritos", "preferencias_regalo", "momentos_vida_negocio", "cambios_familia", "referidos_dados" ]; function getSheet() { return SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Clientes") || SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); } function jsonOutput(obj) { return ContentService .createTextOutput(JSON.stringify(obj)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const sheet = getSheet(); ensureHeaders(sheet); const data = JSON.parse(e.postData.contents); // Edición de un cliente ya existente (viene de consulta.html). if (data.__action === "update" && data.__id) { const rowNum = parseInt(data.__id, 10); if (!rowNum || rowNum < 2 || rowNum > sheet.getLastRow()) { return jsonOutput({ result: "error", message: "Fila no encontrada" }); } const timestampActual = sheet.getRange(rowNum, 1).getValue(); const row = FIELD_ORDER.map(fieldName => { if (fieldName === null) return timestampActual; // conserva el timestamp original return data[fieldName] || ""; }); sheet.getRange(rowNum, 1, 1, row.length).setValues([row]); return jsonOutput({ result: "ok" }); } // Captura normal: agrega una fila nueva. const row = FIELD_ORDER.map(fieldName => { if (fieldName === null) return new Date(); // timestamp automático return data[fieldName] || ""; }); sheet.appendRow(row); return jsonOutput({ result: "ok" }); } catch (err) { return jsonOutput({ result: "error", message: err.toString() }); } } function doGet(e) { try { const sheet = getSheet(); ensureHeaders(sheet); const action = (e.parameter.action || "list"); const lastRow = sheet.getLastRow(); if (lastRow < 2) { return jsonOutput(action === "get" ? { result: "error", message: "No encontrado" } : { result: "ok", clientes: [] }); } const values = sheet.getRange(2, 1, lastRow - 1, HEADERS.length).getValues(); if (action === "get") { const id = parseInt(e.parameter.id, 10); const rowIndex = id - 2; // fila 2 de la hoja = values[0] if (!id || rowIndex < 0 || rowIndex >= values.length) { return jsonOutput({ result: "error", message: "No encontrado" }); } return jsonOutput({ result: "ok", cliente: rowToObject(values[rowIndex], id) }); } // action === "list", con búsqueda opcional por nombre o asesor. const q = (e.parameter.q || "").toLowerCase().trim(); const nombreCol = FIELD_ORDER.indexOf("nombre_completo"); const asesorCol = FIELD_ORDER.indexOf("asesor"); const fechaCol = FIELD_ORDER.indexOf("fecha_captura"); const clientes = []; values.forEach((dataRow, i) => { const nombre = String(dataRow[nombreCol] || ""); const asesor = String(dataRow[asesorCol] || ""); if (q && !nombre.toLowerCase().includes(q) && !asesor.toLowerCase().includes(q)) return; let fecha = dataRow[fechaCol]; if (fecha instanceof Date) { fecha = Utilities.formatDate(fecha, Session.getScriptTimeZone(), "yyyy-MM-dd"); } clientes.push({ id: i + 2, nombre_completo: nombre, asesor: asesor, fecha_captura: fecha || "" }); }); return jsonOutput({ result: "ok", clientes: clientes }); } catch (err) { return jsonOutput({ result: "error", message: err.toString() }); } } // Convierte una fila de valores en un objeto {nombre_campo: valor}, usando FIELD_ORDER. function rowToObject(dataRow, id) { const obj = { id: id }; FIELD_ORDER.forEach((fieldName, i) => { if (fieldName === null) return; // timestamp interno, no se expone como campo editable let value = dataRow[i]; if (value instanceof Date) { value = Utilities.formatDate(value, Session.getScriptTimeZone(), "yyyy-MM-dd"); } obj[fieldName] = (value === null || value === undefined) ? "" : value; }); return obj; } // Crea la fila de encabezados si la hoja está vacía. function ensureHeaders(sheet) { if (sheet.getLastRow() === 0) { sheet.appendRow(HEADERS); sheet.getRange(1, 1, 1, HEADERS.length).setFontWeight("bold"); sheet.setFrozenRows(1); } } // Función opcional para probar manualmente desde el editor de Apps Script. function pruebaManual() { const fakeEvent = { postData: { contents: JSON.stringify({ asesor: "Hector", nivel_lealtad: "Oro", nombre_completo: "Cliente de Prueba" }) } }; doPost(fakeEvent); }
